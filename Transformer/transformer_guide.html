Tokenization

// Breaking language into learnable units

Before a transformer can process text, it must convert it into numbers. Tokenization is this conversion process — splitting text into tokens (subwords, characters, or words) and mapping them to integer IDs.

Modern transformers use Byte-Pair Encoding (BPE) or WordPiece. Rather than splitting on spaces alone, these algorithms learn frequent subword patterns. This handles rare words and typos gracefully — "unhappiness" might become ["un", "happiness"].

Example: "The transformer learns language"
The trans former learns language
→ Token IDs:
464 5425 16843 42 7615

Each token ID is then looked up in an embedding matrix to get a dense vector. An embedding matrix of size vocab_size × d_model maps discrete tokens into continuous vector space where similar tokens cluster together.

💡 GPT-4 uses ~100k vocab tokens. BERT uses ~30k WordPiece tokens. The embedding layer is often the largest single parameter matrix in the model — for GPT-2 (50,257 × 768) that's ~38M parameters just for embeddings.
tokenization.py 
import torch
import torch.nn as nn

# ─── Simple character-level tokenizer ───────────────────────────────────────
class CharTokenizer:
    def __init__(self, text):
        chars = sorted(set(text))
        self.vocab_size = len(chars)
        self.stoi = {ch: i for i, ch in enumerate(chars)}
        self.itos = {i: ch for ch, i in self.stoi.items()}

    def encode(self, text):
        return [self.stoi[c] for c in text]

    def decode(self, ids):
        return ''.join([self.itos[i] for i in ids])


# ─── Token Embedding Layer ───────────────────────────────────────────────────
class TokenEmbedding(nn.Module):
    def __init__(self, vocab_size: int, d_model: int):
        super().__init__()
        # Each token ID → a learnable d_model-dim vector
        self.embedding = nn.Embedding(vocab_size, d_model)
        self.d_model = d_model

    def forward(self, token_ids):
        # Scale embeddings by sqrt(d_model) — original paper trick
        return self.embedding(token_ids) * self.d_model ** 0.5


# ─── Demo ────────────────────────────────────────────────────────────────────
if __name__ == "__main__":
    text = "the transformer learns language"

    # 1. Build vocab and tokenize
    tok = CharTokenizer(text)
    ids = tok.encode(text)
    print(f"Vocab size: {tok.vocab_size}")
    print(f"Token IDs: {ids}")

    # 2. Convert to tensor and embed
    token_tensor = torch.tensor(ids).unsqueeze(0)  # (1, seq_len)

    embed_layer = TokenEmbedding(vocab_size=tok.vocab_size, d_model=64)
    embeddings = embed_layer(token_tensor)

    # Shape: (batch=1, seq_len, d_model=64)
    print(f"Embedding shape: {embeddings.shape}")

    # 3. Decode back
    print(f"Decoded: '{tok.decode(ids)}'")

Positional Encoding

// Teaching the model where each word lives

Transformers process all tokens simultaneously — unlike RNNs, there's no inherent notion of order. Without positional encoding, "dog bites man" and "man bites dog" would produce identical representations.

The original paper uses sinusoidal functions of different frequencies to encode position. For each position pos and each dimension i:

Sinusoidal Positional Encoding
PE(pos, 2i) = sin( pos / 10000^(2i/d_model) )
PE(pos, 2i+1) = cos( pos / 10000^(2i/d_model) )

Even dimensions use sin, odd dimensions use cos. The wavelengths form a geometric progression from 2π to 10000·2π. This lets the model attend to relative positions via linear combinations — a crucial property.

🔑 Modern models (GPT, LLaMA) use learned positional embeddings or RoPE (Rotary Position Embedding) instead of fixed sinusoids. RoPE encodes position directly into attention weights via rotation matrices.
positional_encoding.py 
import torch
import torch.nn as nn
import math


# ─── Sinusoidal Positional Encoding (Original Transformer Paper) ─────────────
class SinusoidalPositionalEncoding(nn.Module):
    def __init__(self, d_model: int, max_len: int = 5000, dropout: float = 0.1):
        super().__init__()
        self.dropout = nn.Dropout(p=dropout)

        # Precompute PE table of shape (max_len, d_model)
        pe = torch.zeros(max_len, d_model)

        # positions: (max_len, 1)
        position = torch.arange(0, max_len).unsqueeze(1).float()

        # div_term: (d_model/2,) — frequency divisors
        div_term = torch.exp(
            torch.arange(0, d_model, 2).float()
            * (-math.log(10000.0) / d_model)
        )

        pe[:, 0::2] = torch.sin(position * div_term)   # even dims → sin
        pe[:, 1::2] = torch.cos(position * div_term)   # odd dims  → cos

        # Add batch dimension: (1, max_len, d_model)
        pe = pe.unsqueeze(0)

        # Register as buffer (not a parameter, but saved with model)
        self.register_buffer('pe', pe)

    def forward(self, x):
        # x: (batch, seq_len, d_model)
        # Add positional signal to token embeddings
        x = x + self.pe[:, :x.size(1), :]
        return self.dropout(x)


# ─── Learned Positional Embedding (GPT-style) ────────────────────────────────
class LearnedPositionalEmbedding(nn.Module):
    def __init__(self, max_len: int, d_model: int):
        super().__init__()
        # Each position has its own learned vector
        self.pos_embedding = nn.Embedding(max_len, d_model)

    def forward(self, x):
        seq_len = x.size(1)
        positions = torch.arange(seq_len, device=x.device)
        return x + self.pos_embedding(positions)


# ─── Demo ────────────────────────────────────────────────────────────────────
if __name__ == "__main__":
    d_model = 64
    seq_len = 10
    batch   = 2

    # Fake token embeddings
    x = torch.randn(batch, seq_len, d_model)

    # Apply sinusoidal PE
    pe = SinusoidalPositionalEncoding(d_model, dropout=0.0)
    x_pe = pe(x)

    print(f"Input  shape: {x.shape}")    # (2, 10, 64)
    print(f"Output shape: {x_pe.shape}")   # (2, 10, 64) — same, positions added

    # Inspect: first two positions, first 4 dims
    print("\nPE values for positions 0–3, dims 0–3:")
    print(pe.pe[0, :4, :4])

Self-Attention

// Every token asks: "who should I listen to?"

Self-attention lets each token look at every other token and decide how much to attend to it. The name "self" means the sequence attends to itself — Queries, Keys, and Values all come from the same input.

❓

Query (Q)

What this token is asking for. "What context do I need to understand myself?"

🔑

Key (K)

What this token offers to others. "Here's what I contain — is it relevant to you?"

💎

Value (V)

The actual information payload. "If you attend to me, here's what you get."

Scaled Dot-Product Attention
Attention(Q, K, V) = softmax( Q·Kᵀ / √d_k ) · V

Scores = Q·Kᵀ ← how much each Q matches each K
Scale = / √d_k ← prevents softmax saturation
Weights = softmax(Scores) ← normalize to sum=1
Output = Weights · V ← weighted sum of Values

The division by √d_k is critical. Without it, large dot products push softmax into regions with tiny gradients, crippling learning. d_k is typically d_model / num_heads.

⚠️ In decoder self-attention, we apply a causal mask — the upper triangle of the score matrix is set to -∞ before softmax. This prevents position i from attending to positions j > i, ensuring autoregressive generation.
self_attention.py 
import torch
import torch.nn as nn
import torch.nn.functional as F
import math


def scaled_dot_product_attention(
    Q: torch.Tensor,
    K: torch.Tensor,
    V: torch.Tensor,
    mask=None
):
    """
    Q, K, V: (batch, seq_len, d_k)
    mask:    (batch, seq_len, seq_len) or (1, seq_len, seq_len)
             True = keep, False = mask out (set to -inf)
    Returns: (batch, seq_len, d_k)
    """
    d_k = Q.size(-1)

    # Step 1: Compute raw attention scores
    # (batch, seq_len, d_k) × (batch, d_k, seq_len) → (batch, seq_len, seq_len)
    scores = torch.matmul(Q, K.transpose(-2, -1)) / math.sqrt(d_k)

    # Step 2: Apply mask (optional) — used for causal/padding masks
    if mask is not None:
        scores = scores.masked_fill(mask == 0, -1e9)

    # Step 3: Softmax over key dimension → attention weights
    attn_weights = F.softmax(scores, dim=-1)   # (batch, seq, seq)

    # Step 4: Weighted sum of Values
    output = torch.matmul(attn_weights, V)       # (batch, seq, d_k)

    return output, attn_weights


class SelfAttention(nn.Module):
    def __init__(self, d_model: int):
        super().__init__()
        self.d_model = d_model

        # Linear projections: input → Q, K, V
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)

    def forward(self, x, mask=None):
        # x: (batch, seq_len, d_model)
        Q = self.W_q(x)   # (batch, seq, d_model)
        K = self.W_k(x)   # (batch, seq, d_model)
        V = self.W_v(x)   # (batch, seq, d_model)

        out, weights = scaled_dot_product_attention(Q, K, V, mask)
        return out, weights


# ─── Demo ────────────────────────────────────────────────────────────────────
if __name__ == "__main__":
    batch, seq_len, d_model = 2, 5, 64
    x = torch.randn(batch, seq_len, d_model)

    # No mask (encoder-style: all tokens see all)
    attn = SelfAttention(d_model)
    out, weights = attn(x)
    print(f"Output shape:  {out.shape}")       # (2, 5, 64)
    print(f"Weights shape: {weights.shape}")   # (2, 5, 5)

    # Causal mask (decoder-style: position i only sees 0..i)
    causal_mask = torch.tril(torch.ones(seq_len, seq_len)).unsqueeze(0)
    out_causal, _ = attn(x, mask=causal_mask)
    print(f"\nCausal output: {out_causal.shape}")  # (2, 5, 64)
    print("\nCausal mask:\n", causal_mask[0].int())

Multi-Head Attention

// Multiple perspectives, parallel attention streams

Single-head attention collapses all reasoning into one subspace. Multi-head attention runs h attention operations in parallel, each on a different linear projection of Q, K, V. Each head can specialize — one might track syntactic structure, another co-reference, another semantics.

Multi-Head Attention
head_i = Attention( Q·W_q_i, K·W_k_i, V·W_v_i )

MultiHead(Q,K,V) = Concat(head_1, ..., head_h) · W_o

where d_k = d_v = d_model / h

For d_model=512 and h=8, each head operates in d_k=64 dimensions. The concatenated output is 8×64=512 dims, then projected through W_o back to d_model. Crucially, the total computation is similar to single-head but the parallel perspectives are vastly richer.

🧠 Research has shown different heads in BERT specialize dramatically: some track syntactic dependencies, some handle coreference (matching "it" to what it refers to), some look at adjacent tokens, some look at sentence boundaries.
multi_head_attention.py 
import torch
import torch.nn as nn
import torch.nn.functional as F
import math


class MultiHeadAttention(nn.Module):
    """
    Multi-Head Attention from 'Attention Is All You Need'.
    Splits d_model into h heads, each with dimension d_k = d_model // h.
    """
    def __init__(self, d_model: int, num_heads: int, dropout: float = 0.1):
        super().__init__()
        assert d_model % num_heads == 0, "d_model must be divisible by num_heads"

        self.d_model   = d_model
        self.num_heads = num_heads
        self.d_k       = d_model // num_heads   # dim per head

        # Single large projection matrices (efficient: one matmul for all heads)
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.W_o = nn.Linear(d_model, d_model)   # output projection

        self.dropout = nn.Dropout(dropout)

    def split_heads(self, x):
        """Reshape (batch, seq, d_model) → (batch, heads, seq, d_k)"""
        batch, seq, _ = x.shape
        x = x.view(batch, seq, self.num_heads, self.d_k)
        return x.transpose(1, 2)  # (batch, heads, seq, d_k)

    def merge_heads(self, x):
        """Reshape (batch, heads, seq, d_k) → (batch, seq, d_model)"""
        batch, _, seq, _ = x.shape
        x = x.transpose(1, 2).contiguous()
        return x.view(batch, seq, self.d_model)

    def forward(self, query, key, value, mask=None):
        # Project inputs to Q, K, V
        Q = self.split_heads(self.W_q(query))  # (B, H, S, d_k)
        K = self.split_heads(self.W_k(key))    # (B, H, S, d_k)
        V = self.split_heads(self.W_v(value))  # (B, H, S, d_k)

        # Scaled dot-product attention across all heads simultaneously
        scores = torch.matmul(Q, K.transpose(-2, -1)) / math.sqrt(self.d_k)

        if mask is not None:
            # Expand mask for heads: (B, 1, S, S) broadcast to (B, H, S, S)
            scores = scores.masked_fill(mask.unsqueeze(1) == 0, -1e9)

        attn_weights = self.dropout(F.softmax(scores, dim=-1))
        attn_output  = torch.matmul(attn_weights, V)   # (B, H, S, d_k)

        # Concatenate heads and project
        out = self.W_o(self.merge_heads(attn_output))  # (B, S, d_model)
        return out, attn_weights


# ─── Demo ────────────────────────────────────────────────────────────────────
if __name__ == "__main__":
    d_model, num_heads = 512, 8
    batch, seq_len     = 2, 10

    mha = MultiHeadAttention(d_model, num_heads)
    x   = torch.randn(batch, seq_len, d_model)

    # Self-attention: Q=K=V=x
    out, w = mha(x, x, x)
    print(f"Output:  {out.shape}")   # (2, 10, 512)
    print(f"Weights: {w.shape}")     # (2, 8, 10, 10) — one per head!

    # Cross-attention: Q from decoder, K/V from encoder
    enc_out = torch.randn(batch, 15, d_model)   # encoder has 15 tokens
    dec_q   = torch.randn(batch, 10, d_model)   # decoder queries 10 tokens
    cross, _ = mha(dec_q, enc_out, enc_out)
    print(f"\nCross-attn: {cross.shape}")   # (2, 10, 512)

Encoder vs Decoder

// Understanding vs Generating

The original transformer is encoder-decoder. But modern models often use just one half. BERT = encoder only. GPT = decoder only. T5 = full encoder-decoder. Each choice shapes what the model is good at.

⬡ Encoder

  • Bidirectional — sees full sequence
  • Self-attention: all → all
  • No causal mask
  • Produces rich contextual representations
  • Used for: classification, NER, embeddings
  • Examples: BERT, RoBERTa, ELECTRA

⬡ Decoder

  • Causal (autoregressive)
  • Self-attention: position i sees 0..i only
  • Causal mask (lower-triangular)
  • Cross-attention to encoder (if present)
  • Used for: text generation, translation
  • Examples: GPT-4, LLaMA, Claude

Each transformer layer (encoder or decoder) follows the same recipe:

Single Encoder Layer
Input x
↓
Multi-Head Self-Attention
↓
Add & LayerNorm → x = LayerNorm(x + Attention(x))
↓
Feed-Forward Network (2 linear layers)
↓
Add & LayerNorm → x = LayerNorm(x + FFN(x))
↓
Output x'
encoder_decoder.py 
import torch
import torch.nn as nn
# (assumes MultiHeadAttention from previous section is imported)


# ─── Feed-Forward Network ─────────────────────────────────────────────────────
class FeedForward(nn.Module):
    """
    FFN(x) = max(0, xW₁ + b₁)W₂ + b₂
    Inner dim is typically 4× d_model (e.g., 512 → 2048 → 512).
    """
    def __init__(self, d_model: int, d_ff: int, dropout: float = 0.1):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(d_model, d_ff),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(d_ff, d_model),
        )

    def forward(self, x):
        return self.net(x)


# ─── Encoder Layer ────────────────────────────────────────────────────────────
class EncoderLayer(nn.Module):
    def __init__(self, d_model, num_heads, d_ff, dropout=0.1):
        super().__init__()
        self.self_attn = MultiHeadAttention(d_model, num_heads, dropout)
        self.ffn       = FeedForward(d_model, d_ff, dropout)
        self.norm1     = nn.LayerNorm(d_model)
        self.norm2     = nn.LayerNorm(d_model)
        self.dropout   = nn.Dropout(dropout)

    def forward(self, x, src_mask=None):
        # Pre-LN variant (more stable): norm before attention
        attn_out, _ = self.self_attn(x, x, x, src_mask)
        x = self.norm1(x + self.dropout(attn_out))   # residual
        x = self.norm2(x + self.dropout(self.ffn(x)))  # residual
        return x


# ─── Decoder Layer ────────────────────────────────────────────────────────────
class DecoderLayer(nn.Module):
    def __init__(self, d_model, num_heads, d_ff, dropout=0.1):
        super().__init__()
        self.self_attn  = MultiHeadAttention(d_model, num_heads, dropout)
        self.cross_attn = MultiHeadAttention(d_model, num_heads, dropout)
        self.ffn        = FeedForward(d_model, d_ff, dropout)
        self.norm1      = nn.LayerNorm(d_model)
        self.norm2      = nn.LayerNorm(d_model)
        self.norm3      = nn.LayerNorm(d_model)
        self.dropout    = nn.Dropout(dropout)

    def forward(self, x, enc_output, src_mask=None, tgt_mask=None):
        # 1. Masked self-attention (causal: can't see future)
        sa_out, _ = self.self_attn(x, x, x, tgt_mask)
        x = self.norm1(x + self.dropout(sa_out))

        # 2. Cross-attention: Q from decoder, K/V from encoder
        ca_out, _ = self.cross_attn(x, enc_output, enc_output, src_mask)
        x = self.norm2(x + self.dropout(ca_out))

        # 3. FFN
        x = self.norm3(x + self.dropout(self.ffn(x)))
        return x

Full Transformer

// Everything wired together

Now we assemble all the pieces. A full seq2seq transformer stacks N encoder layers then N decoder layers, producing logits over the vocabulary at each output position. Below is a complete, working implementation.

transformer.py 
import torch
import torch.nn as nn
import torch.nn.functional as F
import math


# ─────────────────────────────────────────────────────────────────────────────
# COMPLETE TRANSFORMER — all pieces assembled
# ─────────────────────────────────────────────────────────────────────────────

class Transformer(nn.Module):
    def __init__(
        self,
        src_vocab_size: int,
        tgt_vocab_size: int,
        d_model:        int = 512,
        num_heads:      int = 8,
        num_layers:     int = 6,
        d_ff:           int = 2048,
        max_len:        int = 5000,
        dropout:        float = 0.1,
    ):
        super().__init__()

        # ── Embeddings ──────────────────────────────────────────────────────────
        self.src_embed = nn.Embedding(src_vocab_size, d_model)
        self.tgt_embed = nn.Embedding(tgt_vocab_size, d_model)
        self.pos_enc   = SinusoidalPositionalEncoding(d_model, max_len, dropout)
        self.scale     = math.sqrt(d_model)

        # ── Encoder stack ──────────────────────────────────────────────────────
        self.encoder_layers = nn.ModuleList([
            EncoderLayer(d_model, num_heads, d_ff, dropout)
            for _ in range(num_layers)
        ])
        self.encoder_norm = nn.LayerNorm(d_model)

        # ── Decoder stack ──────────────────────────────────────────────────────
        self.decoder_layers = nn.ModuleList([
            DecoderLayer(d_model, num_heads, d_ff, dropout)
            for _ in range(num_layers)
        ])
        self.decoder_norm = nn.LayerNorm(d_model)

        # ── Output projection ──────────────────────────────────────────────────
        self.output_proj = nn.Linear(d_model, tgt_vocab_size)

        self._init_weights()

    def _init_weights(self):
        for p in self.parameters():
            if p.dim() > 1:
                nn.init.xavier_uniform_(p)

    def make_causal_mask(self, size, device):
        """Lower-triangular mask for autoregressive decoding."""
        return torch.tril(torch.ones(1, size, size, device=device))

    def encode(self, src, src_mask=None):
        x = self.pos_enc(self.src_embed(src) * self.scale)
        for layer in self.encoder_layers:
            x = layer(x, src_mask)
        return self.encoder_norm(x)

    def decode(self, tgt, enc_out, src_mask=None, tgt_mask=None):
        x = self.pos_enc(self.tgt_embed(tgt) * self.scale)
        for layer in self.decoder_layers:
            x = layer(x, enc_out, src_mask, tgt_mask)
        return self.decoder_norm(x)

    def forward(self, src, tgt, src_mask=None):
        tgt_mask = self.make_causal_mask(tgt.size(1), tgt.device)
        enc_out  = self.encode(src, src_mask)
        dec_out  = self.decode(tgt, enc_out, src_mask, tgt_mask)
        return self.output_proj(dec_out)   # (batch, tgt_len, tgt_vocab)


# ─── Demo: count parameters and run forward pass ─────────────────────────────
if __name__ == "__main__":
    model = Transformer(
        src_vocab_size=10000,
        tgt_vocab_size=10000,
        d_model=512,
        num_heads=8,
        num_layers=6,
        d_ff=2048,
    )

    n_params = sum(p.numel() for p in model.parameters())
    print(f"Parameters: {n_params/1e6:.1f}M")  # ~44M like original paper

    # Fake batch: 2 sequences of 20 src tokens, 15 tgt tokens
    src = torch.randint(0, 10000, (2, 20))
    tgt = torch.randint(0, 10000, (2, 15))

    logits = model(src, tgt)
    print(f"Output logits: {logits.shape}")  # (2, 15, 10000)

    # Training: cross-entropy loss
    labels  = torch.randint(0, 10000, (2, 15))
    loss    = F.cross_entropy(logits.view(-1, 10000), labels.view(-1))
    print(f"Loss: {loss.item():.4f}")

    # Autoregressive greedy generation
    def greedy_generate(model, src, max_new_tokens=20, bos_id=1, eos_id=2):
        model.eval()
        with torch.no_grad():
            enc_out  = model.encode(src)
            gen      = torch.tensor([[bos_id]], device=src.device)

            for _ in range(max_new_tokens):
                tgt_mask = model.make_causal_mask(gen.size(1), src.device)
                dec_out  = model.decode(gen, enc_out, tgt_mask=tgt_mask)
                logits   = model.output_proj(dec_out[:, -1, :])
                next_tok = logits.argmax(-1, keepdim=True)
                gen      = torch.cat([gen, next_tok], dim=1)
                if next_tok.item() == eos_id:
                    break
        return gen

    gen = greedy_generate(model, src[:1])
    print(f"\nGenerated token IDs: {gen[0].tolist()}")

Quick recap of all dimensions:
x flows as (batch, seq_len, d_model) throughout. The attention mechanism internally works with (batch, heads, seq, d_k). The FFN expands to d_ff = 4 × d_model then contracts back. Output logits are (batch, seq_len, vocab_size).
